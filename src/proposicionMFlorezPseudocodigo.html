Algoritmo Proposición_MFlorez 
    
    Escribir "La suma de los elementos a[1,1], a[1,n], a[n,1], a[n^2,n^2]"
    Escribir "de una matriz de dimensión (n * n) con los números"
    Escribir "naturales mayores a 1 consecutivos y ordenados por renglon"
    Escribir "y columna de la matriz nunca será igual a un número PRIMO"
    Escribir ""
    
    Escribir "Ingrese un número que define la matriz (n*n)"
    
    Leer dimension_matriz // Dimension matriz (n*n)
    
    Si dimension_matriz > 1 Entonces
        Definir vertice_a Como Entero
        Definir vertice_b Como Entero
        Definir vertice_c Como Entero
        Definir vertice_d Como Entero
        
        vertice_a←dimension_matriz - (dimension_matriz - 1);                         //Extremo superior izquierdo de la matriz
        vertice_b←dimension_matriz;                                                 //Extremo superior derecho de la matriz
        vertice_c←dimension_matriz * dimension_matriz - (dimension_matriz - 1);        //Extremo inferior izquierdo de la matriz
        vertice_d←dimension_matriz * dimension_matriz;                                 //Extremo inferior derecho de la matriz
        
        suma_verfices = vertice_a + vertice_b + vertice_c + vertice_d;              // Suma vertices
        
        // Validar si Primo
        contador←0;
        Para i←1 Hasta suma_verfices Hacer
            si suma_verfices%i = 0 Entonces
                contador←contador + 1;
            FinSi
        FinPara
        
        Escribir "Los vertices de la matriz (" dimension_matriz, "*", dimension_matriz, ") son: ", vertice_a, ", ", vertice_b, ", ",vertice_c, ", ",vertice_d
        Escribir "La suma de las vertices de la matriz es :", suma_verfices;
        
        // ¿Es Primo?
        si contador=2 Entonces
            Escribir "La Suma da un número primo";
        SiNo
            Escribir "La Suma NO es un número primo"
        FinSi
        
    SiNo
        Escribir "Dimension no valida."
    Fin Si
    
FinAlgoritmo
